// main.c - BLE 應用程式進入點
#include "ble_lib_api.h"
#include "FreeRTOS.h"
#include "task.h"

int main(void) {
    // 1. 平台初始化
    platform_init(0);
    setup_heap();
    
    // 2. 初始化 BLE Controller
    ble_controller_init(configMAX_PRIORITIES - 3, 1);
    
    // 3. 啟動 FreeRTOS
    vTaskStartScheduler();
    
    return 0;
}

// 進入 Deep Sleep
void enter_deep_sleep(void) {
    // 設定喚醒源
    hal_pmu_set_wakeup_source(PMU_WAKEUP_RTC);
    hal_rtc_set_alarm(60);  // 60 秒後喚醒
    
    // 關閉不需要的周邊
    hal_gpio_set_all_analog();
    
    // 進入 Deep Sleep
    hal_pmu_enter_deep_sleep();
}

// 預期待機電流: < 5μA
