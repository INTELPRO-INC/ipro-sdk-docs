// main.c - BLE Application Entry Point
#include "ble_lib_api.h"
#include "FreeRTOS.h"
#include "task.h"

int main(void) {
    // 1. Platform initialization
    platform_init(0);
    setup_heap();
    
    // 2. Initialize BLE Controller
    ble_controller_init(configMAX_PRIORITIES - 3, 1);
    
    // 3. Start FreeRTOS
    vTaskStartScheduler();
    
    return 0;
}

// Enter Deep Sleep
void enter_deep_sleep(void) {
    // Set wake-up source
    hal_pmu_set_wakeup_source(PMU_WAKEUP_RTC);
    hal_rtc_set_alarm(60);  // Wake up after 60 seconds
    
    // Disable unnecessary peripherals
    hal_gpio_set_all_analog();
    
    // Enter Deep Sleep
    hal_pmu_enter_deep_sleep();
}
