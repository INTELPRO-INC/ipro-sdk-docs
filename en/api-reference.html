API Reference

This page provides a reference for the main IPRO SDK APIs. The BLE API uses Zephyr-style bt_* callback-based APIs, and the HAL API provides low-level hardware access. For complete API documentation, please refer to the header file comments in the source code.

BLE Core Initialization

The BLE subsystem uses Zephyr-style bt_* APIs. bt_enable() must be called before any BLE operations.

bluetooth/bluetooth.h 
/** Enable the Bluetooth subsystem
 *  @param cb  Completion callback, NULL for synchronous enable
 *  @return 0 on success, negative error code on failure
 */
int bt_enable(bt_ready_cb_t cb);

/** Set the Bluetooth device name
 *  @param name  New name string
 *  @return 0 on success, negative error code on failure
 */
int bt_set_name(const char *name);

/** Get the Bluetooth device name
 *  @return Device name string
 */
const char *bt_get_name(void);

Usage Example

BLE Initialization Example 
#include <bluetooth/bluetooth.h>

static void bt_ready(int err)
{
    if (err) {
        printk("BLE init failed (err %d)\n", err);
        return;
    }
    printk("BLE initialized\n");
    // Can start advertising or scanning
}

void main(void)
{
    int err = bt_enable(bt_ready);
    if (err) {
        printk("bt_enable failed (err %d)\n", err);
    }
}

Usage Example

Advertising Example 
static const struct bt_data ad[] = {
    BT_DATA_BYTES(BT_DATA_FLAGS, (BT_LE_AD_GENERAL | BT_LE_AD_NO_BREDR)),
    BT_DATA_BYTES(BT_DATA_UUID16_ALL, 0x0d, 0x18),  // Heart Rate Service

Connection Event Callbacks

{
    if (err) {
        printk("Connection failed (err 0x%02x)\n", err);
    } else {
        printk("Connected\n");
    }
}

static void disconnected(struct bt_conn *conn, u8_t reason)
{
    printk("Disconnected (reason 0x%02x)\n", reason);

static struct bt_conn_cb conn_callbacks = {

    .disconnected = disconnected,
};

// Register callbacks
bt_conn_cb_register(&conn_callbacks);

GATT Server

bluetooth/gatt.h - Server API 
/** Register a GATT service
 *  @param svc  Service definition structure
 *  @return 0 on success, negative error code on failure
 */
int bt_gatt_service_register(struct bt_gatt_service *svc);

/** Send a GATT Notification
 *  @param conn   Connection object (NULL = all subscribed connections)
 *  @param params Notification parameters (attribute, data, length, callback)
 *  @return 0 on success, negative error code on failure
 */
int bt_gatt_notify_cb(struct bt_conn *conn,
                      struct bt_gatt_notify_params *params);

/** Simplified Notify (no callback)
 *  @param conn  Connection object
 *  @param attr  Attribute pointer
 *  @param data  Data pointer
 *  @param len   Data length
 *  @return 0 on success, negative error code on failure
 */
int bt_gatt_notify(struct bt_conn *conn, const struct bt_gatt_attr *attr,
                   const void *data, u16_t len);

/** Send a GATT Indication
 *  @param conn   Connection object
 *  @param params Indication parameters
 *  @return 0 on success, negative error code on failure
 */
int bt_gatt_indicate(struct bt_conn *conn,
                     struct bt_gatt_indicate_params *params);

Service Definition Example

Custom GATT Service 
// Custom service UUID
static struct bt_uuid_128 my_svc_uuid = BT_UUID_INIT_128(
    0x01, 0x23, 0x45, 0x67, 0x89, 0xab, 0xcd, 0xef,
    0x01, 0x23, 0x45, 0x67, 0x89, 0xab, 0xcd, 0xef);

// Read callback
static ssize_t read_value(struct bt_conn *conn,
                          const struct bt_gatt_attr *attr,
                          void *buf, u16_t len, u16_t offset)
{
    return bt_gatt_attr_read(conn, attr, buf, len, offset,
                             &my_value, sizeof(my_value));
}

// Service definition
BT_GATT_SERVICE_DEFINE(my_svc,
    BT_GATT_PRIMARY_SERVICE(&my_svc_uuid),
    BT_GATT_CHARACTERISTIC(&my_char_uuid.uuid,
        BT_GATT_CHRC_READ | BT_GATT_CHRC_NOTIFY,
        BT_GATT_PERM_READ,
        read_value, NULL, NULL),
    BT_GATT_CCC(NULL, BT_GATT_PERM_READ | BT_GATT_PERM_WRITE),
);

 *  @param pin    Pin number
 *  @param cbfun  Interrupt callback function
 */
void gpio_attach_irq(uint32_t pin, void (*cbfun)(uint32_t pin));

/** Enable/disable GPIO interrupt
 *  @param pin     Pin number
 *  @param enabled 1=enable, 0=disable
 */
void gpio_irq_enable(uint32_t pin, uint8_t enabled);
