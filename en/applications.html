CLI Command Examples

Common CLI Commands 
# System Information
help                  # Show all commands
ble status            # BLE status

# Advertising Control
ble adv start         # Start advertising
ble adv stop          # Stop advertising
ble adv data          # Show advertising data

# Connection Management
ble conn list         # List connections
ble conn info 0       # Connection details
ble conn disconnect 0 # Disconnect

# Service Operations
ble hrs update 80     # Update heart rate
ble bas update 75     # Update battery level
ble uart send "Hello" # Send UART data

main.c

#include <zephyr/logging/log.h>

LOG_MODULE_REGISTER(ipro_zephyr, LOG_LEVEL_INF);

int main(void)
{
    LOG_INF("IPRO Zephyr Demo Started!");
    LOG_INF("Board: %s", CONFIG_BOARD);

Code Example

GPIO Usage Example 
#include "ipro_hal_gpio.h"

void gpio_example(void)
{
    // Configure GPIO as output
    ipro_gpio_config_t config = {
        .mode = IPRO_GPIO_MODE_OUTPUT,
        .pull = IPRO_GPIO_PULL_NONE,
    };
    ipro_hal_gpio_init(GPIO_PIN_0, &config);
    
    // Set output
    ipro_hal_gpio_write(GPIO_PIN_0, 1);
    
    // Read input
    uint8_t value = ipro_hal_gpio_read(GPIO_PIN_1);
}

Creating New Applications

Follow these steps to create your own application.

Step 1: Create Directory

Create Application Directory 
mkdir -p apps/my_app/include

Step 2: Create CMakeLists.txt

apps/my_app/CMakeLists.txt 
cmake_minimum_required(VERSION 3.13)

# Set application name
set(APP_NAME my_app)

# Include SDK build system
include(${IPRO_SDK_BASE}/cmake/extension.cmake)

# Add source files
target_sources(app PRIVATE
    main.c
)

# Add include directories
target_include_directories(app PRIVATE
    include
)

Step 3: Create proj.conf

apps/my_app/proj.conf 
# Basic configuration
CONFIG_LOG_DEFAULT_LEVEL=3

# Enable components as needed
# CONFIG_IPRO_BLE_LIB_ENABLE=y
# CONFIG_LWIP_ENABLE=y

Step 4: Create Main Program

apps/my_app/main.c 
#include <stdio.h>
#include "FreeRTOS.h"
#include "task.h"
#include "ipro_log.h"

#define TAG "MY_APP"

void app_task(void *params)
{
    IPRO_LOG_INFO(TAG, "My App Started!");
    
    while (1) {
        IPRO_LOG_INFO(TAG, "Running...");
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

int main(void)
{
    // System initialization
    system_init();
    
    // Create application task
    xTaskCreate(app_task, "app", 4096, NULL, 5, NULL);
    
    // Start scheduler
    vTaskStartScheduler();
    
    return 0;
}

Step 5: Build

Build New Application 
./build_freertos.sh my_app
